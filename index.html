Code Examples

Motif search

let seq = DNA("ACGTACGTGATACAAGT");
let hits = find_motif(seq, "ACGT");
print(hits)  // -> [0, 4]

VCF slicing

let vcf = VCF::open("cohort.vcf.gz");
let snps = vcf.filter(chr: "7", pos: 117199646..117199900);
export(snps, to: "out/snps.tsv")
